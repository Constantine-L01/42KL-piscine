## ft_sort_int_tab.c
```c
void    ft_sort_int_tab(int *tab, int size)
{
    int     unSorted;
    int     iterStartPos;
    int     temp;

    unSorted = -1;
    while (unSorted++ < size - 1)
    {
        iterStartPos = unSorted;
        while (iterStartPos++ < size - 1)
        {
            if (tab[iterStartPos] < tab[unSorted])
            {
                temp = tab[iterStartPos];
                tab[iterStartPos] = tab[unSorted];
                tab[unSorted] = temp;
            }
        }
    }
}
```
