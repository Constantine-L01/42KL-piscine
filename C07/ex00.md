# ft_strdup.c

```c
#include <stdlib.h>

char* ft_strdup(char* src)
{
    int i = 0;
    char* dest;
    int count = 0;

    while(src[count]){
        count++;
    }
    if((dest = malloc(count * sizeof(*dest) + 1)) == 0){
        return 0;
    }
    while(src[i]){
        dest[i] = src[i];
        i++;
    }
    dest[i] = '\0';
    return dest;
}
```
