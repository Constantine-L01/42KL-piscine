# ft_ultimate_range.c

```c
#include <stdlib.h>

int ft_ultimate_range(int** range, int min, int max){
    int i = 0;

    if(min >= max){
        *range = NULL;
        return 0;
    }
    if(!(*range = malloc((max - min) * sizeof(i)))){
        return -1;
    }
    while(min + i < max){
        (*range)[i] = min + i;
        i++;
    }
    return i;
}
```
