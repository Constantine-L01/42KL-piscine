## ft_strstr.c
```c
char* ft_strstr(char* str, char* to_find)
{
    int i = 0;
    int j = 0;

    if(to_find[0] == '\0')
    {
        return str;
    }
    while(str[i])
    {
        j = 0;
        while(str[i + j] && str[i + j] == to_find[j])
        {
            if(to_find[j + 1] == '\0')
            {
                return &str[i];
            }
            j++;
        }
        i++;
    }
    return 0;
}
```
