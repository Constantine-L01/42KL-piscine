## ft_print_combn.c

```c++
#include <unistd.h>

void fEncapWrite(char c)
{
  write(1, &c, 1);
}

void customPrint(int* currNum, int n)
{
  int i;
  i = 0;
  while(++i < n)
    if (currNum[i - 1] >= currNum[i])
      return ;
  i = -1;
  while (++i < n)
    fEncapWrite(currNum[i] + 48);
  if (currNum[0] < (10 - n))
  {
    fEncapWrite(',');
    fEncapWrite(' ');
  }
}

void ft_print_combn(int n)
{
  int test;

  test = 999;
  write(1, &test, sizeof(test));
  int i;
  int currNum[10];

  if (n > 10 || n < 0)
    return ;
  i = -1;
  while (++i < n)
  {
    currNum[i] = i;
  }
  while (currNum[0] <= (10 - n) && n >= 1)
  {
    customPrint(currNum, n);
    if (n == 10)
      break ;
    currNum[n - 1]++;
    i = n;
    while (i && n > 1)
    {
      i--;
      if (currNum[i] > 9)
      {
        currNum[i - 1]++;
        currNum[i] = 0;
      }
    }
  }
}
```
