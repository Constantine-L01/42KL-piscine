## ft_putnbr.c

```c++
#include <unistd.h>

void customPrint(char c)
{
  write(1, &c, 1);
}

void ft_putnbr(int nb)
{
  if (nb < 0)
  {
    nb = -nb;
    customPrint('-');
  }
  if (nb >= 10)
  {
    ft_putnbr(nb / 10);
    ft_putnbr(nb % 10);
  }
  else
    customPrint(nb + '0');
}
```
