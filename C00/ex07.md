## ft_putnbr.c

```c++
#include <unistd.h>

void    customPrint(char c)
{
    write(1, &c, 1);
}

void    ft_putnbr(int   nb)
{
    char    array[10];
    int     i;
    int     isPositive;

    i = 0;
    if (nb > 0)
        isPositive = 1;
    else
        isPositive = -1;
    while (0 < nb * isPositive)
    {
        array[i] = nb % 10 * isPositive;
        nb /= 10;
        i++;
    }
    if (isPositive == -1)
        customPrint('-');
    while (--i >= 0)
    {
        customPrint(array[i] + '0');
    }
}
```
