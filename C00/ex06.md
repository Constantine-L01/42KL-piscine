# ft_print_comb2.c

```c
#include <unistd.h>

void customPrint(char c)
{
  write(1, &c, 1);
}

void ft_print_comb2(void)
{
  char firstP;
  char secP;

  firstP = -1;
  while (firstP++ < 99)
  {
    secP = firstP;
    while (secP++ < 99)
    {
      if (!(firstP == 0 && secP == 1))
      {
        customPrint(',');
        customPrint(' ');
      }
      customPrint(firstP / 10 + '0');
      customPrint(firstP % 10 + '0');
      customPrint(' ');
      customPrint(secP / 10 + '0');
      customPrint(secP % 10 + '0');
    }
  }
}
```
