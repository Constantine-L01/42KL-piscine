# ft_is_prime.c

```c
int isPrime(unsigned int n, unsigned int i){
    if(n % i == 0){
        return 0;
    }
    else if(i * i > n){
        return 1;
    }
    return isPrime(n, i + 1);
}

int ft_is_prime(int nb){
    if(nb < 2){
        return 0;
    }
    else if(nb == 2){
        return 1;
    }
    return isPrime((unsigned int)nb, 2);
}
```
