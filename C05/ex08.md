# ft_ten_queens_puzzle.c

```c
void printBoard(int* board){
    int i = 0;
    char c ;

    while(i < 10){
        c = board[i] + '0';
        write(1, &c, 1);
        i++;
    }
    write(1, "\n", 1);
}

int abs(int a, int b){
    if(a >= b){
        return a - b;
    }
    else{
        return b - a;
    }
}


int isValid(int queenID, int* board){
    int i = 0;
    while(i < queenID){
        if(board[queenID] == board[i] || abs(board[queenID], board[i]) == (queenID - i)){
            return 0;
        }
        i++;
    }
    return 1;
}

void solver(int queenID, int* board, int* noOfSol){
    board[queenID] = 0;
    while(board[queenID] < 10){
        if(isValid(queenID, board)){
            if(queenID == 9){
                printBoard(board);
                (*noOfSol)++;
            }
            else{
                solver(queenID + 1, board, noOfSol);
            }
        }
        board[queenID]++;

    }
}

int ft_ten_queens_puzzle(void){
    int board[10];
    int queenID = 0;
    int noOfSol = 0;

    solver(queenID, board, &noOfSol);
    return noOfSol;
}
```
