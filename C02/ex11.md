#include <unistd.h>

void ft_putchar(char c)
{
    write(1, &c, 1);
}

void ft_print_hex(char c, char* hexCode)
{
    if(c > 16)
    {
        ft_print_hex(c / 16, hexCode);
        ft_print_hex(c % 16, hexCode);
    }
    else
    {
        ft_putchar(hexCode[(int)c]);
    }
}

void ft_putstr_non_printable(char* str)
{
    char* hexCode = "0123456789abcdef";

    while(*str != '\0')
    {
        if(' ' <= *str && *str <= '~')
        {
            ft_putchar(*str);
        }
        else
        {
            ft_putchar('\\');
            if(*str < 16)
            {
                ft_putchar('0');
            }
            ft_print_hex(*str, hexCode);
        }
        str++;
    }
}
