int isFirst(char* c)
{
    char* prev = c - 1;
    if(('a' <= *prev && *prev <= 'z') || ('A' <= *prev && *prev <= 'Z'))
    {
        return 0;
    }
    return 1;
}

char* ft_strcapitalize(char* str)
{
    char* ret = str;

    while(*str)
    {
        if(str == ret || isFirst(str))
        {
            if('a' <= *str && *str <= 'z')
            {
                *str = *str - 'a' + 'A';
            }
        }
        else if('A' <= *str && *str <= 'Z')
        {
            *str = *str - 'A' + 'a';
        }
        str++;
    }
    return ret;
}
