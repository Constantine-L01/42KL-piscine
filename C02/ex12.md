#include <unistd.h>

void ftPutChar(char c)
{
    write(1, &c, 1);
}

void ftPrintAddr(void* addr)
{
    char* hexCode = "0123456789abcdef";
    char arrAdd[16];
    long addrVal = (long)addr;

    for(int i = 0; i < 16; i++)
        arrAdd[i] = '0';

    int i = 0;
    while(addrVal > 0)
    {
        arrAdd[i] = hexCode[addrVal % 16];
        addrVal /= 16;
        i++;
    }
    i = 15;
    while(i >= 0)
    {
        ftPutChar(arrAdd[i]);
        i--;
    }
    ftPutChar(':');
    ftPutChar(' ');
}

void ftPrintString(void* addr, int size)
{
    char* pC = (char*)addr;
    int i = 0;
    while(i < size)
    {
        if(' ' <= pC[i] && pC[i] <= '~')
        {
            ftPutChar(pC[i]);
        }
        else
        {
            ftPutChar('.');
        }
        i++;
    }
}

void ftPrintHexa(void* addr, int size)
{
    char* hexCode = "0123456789abcdef";
    char* pC = (char*)addr;
    int i = 0;

    while(i < size)
    {
        ftPutChar(hexCode[pC[i] / 16]);
        ftPutChar(hexCode[pC[i] % 16]);
        if(i % 2)
        {
            ftPutChar(' ');
        }
        i++;
    }
    while(i < 16)
    {
        ftPutChar(' ');
        ftPutChar(' ');
        if(i % 2)
        {
            ftPutChar(' ');
        }
        i++;
    }
}

void ftPrintLine(void* addr, int size)
{
    ftPrintAddr(addr);
    ftPrintHexa(addr, size);
    ftPrintString(addr, size);
}

void* ft_print_memory(void* addr, unsigned int size)
{
    unsigned int charToPrint;
    unsigned int offset = 0;

    while(size > 0)
    {
        if(size < 16)
        {
            charToPrint = size;
        }
        else
        {
            charToPrint = 16;
        }
        size -= charToPrint;
        ftPrintAddr(addr + offset);
        ftPrintHexa(addr + offset, charToPrint);
        ftPrintString(addr + offset, charToPrint);
        ftPutChar('\n');
        offset += 16;
    }
    return addr;
}
